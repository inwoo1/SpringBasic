
create table score)
    num number(10) primary key,
    name varchar2(30),
    kor number(3),
    eng number(3),
    math number(3)
);

create SEQUENCE score_seq Increment 1 start with 1 nocache;



create table board(
    bno number(10) primary key,
    name varchar2(30),
    title varchar2(30),
    content varchar2(30)


);

create SEQUENCE board_seq INCREMENT by 1 start with 1 nocache;
